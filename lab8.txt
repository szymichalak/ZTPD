-- zadanie 2

WHERE UPPER(TEKST) LIKE '%PESYMISTA%' AND UPPER(TEKST) LIKE '%OPTYMISTA%';

-- zadanie 3

-- zadanie 9

FROM CYTATY
WHERE CONTAINS(TEKST, 'życi%', 1) > 0;

FROM CYTATY
WHERE CONTAINS(TEKST, 'życi%', 1) > 0 AND ROWNUM <= 1
